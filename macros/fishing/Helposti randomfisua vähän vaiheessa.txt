{
HALT
}

alku:
finditem %pole
set #lObjectID #findid
set %random1 #random % 5 + 2
set %random2 #random % 5 + 2
set %random3 #random % 5 + 2
set #LTargetX %random1 + #CharPosX
set #LTargetY %random2 + #CharPosY
set #LtargetZ %random3 + #charposZ
event Macro 17
target
event macro 22 0
goto zekki


zekki:
set %jrnl #jindex
while #true
   {
   if #jindex > %jrnl
      {
      set %jrnl %jrnl + 1
      scanjournal %jrnl
      if Pyydystät in #journal
      goto alku
      {
      if Tästä_kohtaa_ei_tunnu_kaloja_olevan in #journal
      goto alku
      {
      if Kala_pääsi_karkuun in #journal
      goto alku
      {
      if Ongit_ylös_vanhan_pullon in #journal
      goto alku
      }
      else
      {
      goto zekki
      }
      }
      }
      }
      }
