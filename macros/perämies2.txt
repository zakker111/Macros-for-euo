menu show
set #menubutton N/A

MainLoop:
  wait 5
  if #menubutton = N/A
     goto MainLoop
  if #menubutton = closed
     gosub closed
  if _ in #menubutton
  {
     gosub replace_
     goto MainLoop
  }
  gosub boatcommand
  set #menubutton N/A
goto MainLoop


sub boatcommand
  set %temp #menubutton
  str lower %temp
  if #menubutton = Seilaa
  {
    ;jos ei numero,niin sitten karttapisteeseen 1
    menu GetNum piste 1
    msg #strres #menuRes $
    return
  }
  if #menubutton = Kartta
  {   
    msg Aseta kurssi $
    gosub kohdista_karttaan
    return
  }
  if #menubutton = Muokkaa
  {   
    gosub muokkaa_karttaa
    return
  }
  msg #strres $

return

sub replace_
  str pos #menubutton _
  set %strlen #strres - 1
  set %strlen2 #strres + 1
  str left #menubutton %strlen
  set %_1 #strres
  str mid #menubutton %strlen2 10
  set #menubutton %_1 , #spc , #strres
return

sub kohdista_karttaan
  set #LTargetID %Kartta
  target
  key F8
return

sub muokkaa_karttaa
  set #LObjectID %Kartta
  key F5
return

sub CLOSED
    halt
return
